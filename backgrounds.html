Background CSS

                  
/* Backgrounds */

.shr_bg_primary_to_sunlight {
    /* Gradients/Faded/PrimaryToSunlightFaded */
    background: linear-gradient(262.47deg, #FEF9EB 11.14%, #E9F2FF 101.49%);
}

.shr_bg_primary_lt3 {
    background: var(--shr_primary_blue_lt3);
}

.shr_bg_success_to_primary {
    background: linear-gradient(267.56deg, #EFFBF2 -17.28%, #E9F2FF 70.25%);
    transform: matrix(-1, 0, 0, 1, 0, 0);
}

.shr_bg_success {
    background: #F4FBF4;
    transform: matrix(-1, 0, 0, 1, 0, 0);
}

.shr_bg_magenta_to_primary {
    /* Gradients/Faded/MagentaToPrimaryFaded */
    background: linear-gradient(236.96deg, #F9F3FA 28.56%, #ECF4FF 90.53%);
    transform: matrix(-1, 0, 0, 1, 0, 0);
}

.shr_bg_dark {
    /* Dark/LT-3 */
    background: #F5F5F5;
    transform: matrix(-1, 0, 0, 1, 0, 0);
}
