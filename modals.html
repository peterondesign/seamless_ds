<!-- SHR Modal Script -->
<script>
    // Get the modal
    let shr_default_modal = document.getElementById("shr_default_modal");

    // Get the button that opens the modal
    let shr_trigger_default_modal = document.getElementById("shr_trigger_default_modal");

    // Get the <span> element that closes the modal
    let shr_default_modal_close = document.getElementsByClassName("shr_default_modal_close")[0];
    let shr_default_button_close = document.getElementById("shr_default_button_close");


    // When the user clicks the button, open the modal 
    shr_trigger_default_modal.onclick = function() {
        shr_default_modal.style.display = "block";
    }

    // When the user clicks on <span> (x), close the modal
    shr_default_modal_close.onclick = function() {
        shr_default_modal.style.display = "none";
    }

    // When the user clicks anywhere outside of the modal, close it
    window.onclick = function(event) {
        if (event.target == shr_default_modal) {
            shr_default_modal.style.display = "none";
        }
    }
</script>
