<script>
    let badges = document.getElementById('shr_badge_wrapper'),
        div_creator = document.createElement('div'),
        clone;
    ['Awaiting response', 'Approved', 'Mark', 'Rejected', 'Information', '99+'].forEach(function(badge_type, index) {
        clone = div_creator.cloneNode();
        clone.innerHTML = '<div class="shr_badge shr_badge_blue" style="margin-right: 10px; margin-bottom: 15px;"><span>' + badge_type + '</span></div>';
        badges.appendChild(clone);
    });
</script>
